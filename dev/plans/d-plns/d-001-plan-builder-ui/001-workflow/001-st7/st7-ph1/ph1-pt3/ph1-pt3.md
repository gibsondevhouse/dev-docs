## Exact Requirements

1. Add to `src-python/services/plan_builder.py`:

   ```python
   import re

   def update_master_plan(plans_dir: Path, plan: Plan, plan_folder_name: str, type_folder: str) -> None:
       """Insert a new row into the Plan Registry table in master-plan.md."""
       master_path = plans_dir / "master-plan.md"
       if not master_path.exists():
           return
       content = master_path.read_text(encoding="utf-8")
       new_row = (
           f"| {plan.number} | {plan.type} | {plan.name.replace('-', ' ').title()} "
           f"| Active | `{type_folder}/{plan_folder_name}/` | Generated by Plan Builder UI |"
       )
       # Insert before the status key line (after the last table row)
       content = content.replace(
           "**Status key:**",
           f"{new_row}\n\n**Status key:**",
       )
       # Update plan counts
       content = re.sub(r"- \*\*Total plans:\*\* \d+", f"- **Total plans:** 1", content)
       content = re.sub(r"- \*\*Active:\*\* \d+", "- **Active:** 1", content)
       master_path.write_text(content, encoding="utf-8")

   def update_active_plan(plans_dir: Path, plan: Plan, plan_folder_name: str, type_folder: str) -> None:
       """Replace the placeholder section in active-plan.md with this plan's details."""
       active_path = plans_dir / "active-plan.md"
       if not active_path.exists():
           return
       content = active_path.read_text(encoding="utf-8")
       new_section = f"""### Plan {plan.number} — {plan.name.replace("-", " ").title()}

   - **Type:** {"Development" if plan.type == "d" else "Polishing" if plan.type == "p" else "Refactoring"}
   - **Plan folder:** `{type_folder}/{plan_folder_name}/`
   - **Main brief:** `{type_folder}/{plan_folder_name}/{plan.number}.md`
   - **Pick up at:** Stage 1 → Phase 1 → Part 1
   - **Last completed:** Plan created via Plan Builder UI
   - **Status:** Ready to start

   **Blockers:** None
   """
       # Replace the placeholder section
       content = content.replace(
           "### Plan [NNN] — [Plan Name]",
           new_section,
           1,
       )
       active_path.write_text(content, encoding="utf-8")
   ```
