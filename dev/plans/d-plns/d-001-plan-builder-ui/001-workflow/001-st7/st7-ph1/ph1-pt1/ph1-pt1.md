1. Add to `src-python/services/plan_builder.py` a function that generates all plan files as a dict of `{relative_path: content}`:

   ```python
   from pathlib import Path

   def generate_plan_files(plan: Plan) -> dict[str, str]:
       """Returns {relative_path: markdown_content} for every file in the plan."""
       n = plan.number
       files: dict[str, str] = {}

       # Root folder name
       root = f"{plan.type}-{n}-{plan.name}.md"

       # Master brief
       stages_map = "\n".join(
           f"├── {n}-{st.id}/\n" + "\n".join(
               f"│   ├── {ph.id}/\n" + "\n".join(
                   f"│   │   └── {pt.id}/"
                   for pt in ph.parts
               )
               for ph in st.phases
           )
           for st in plan.stages
       )
       files[f"{root}/{n}.md"] = f"""# Plan {n} — {plan.name.replace("-", " ").title()}

   ## Agent Briefing

   **Plan type:** {plan.type.upper()}
   **Plan number:** {n}
   **Plan name:** {plan.name}

   ---

   ## What This Plan Does

   {plan.description}

   ---

   ## Workflow Map

   ```
   {n}-workflow/
   {stages_map}
   ```

   ---

   ## Stages at a Glance

   | Stage | Name | Purpose | Status |
   |-------|------|---------|--------|
   """ + "\n".join(
           f"| {st.id.upper()} | {st.name} | {st.goal} | Not started |"
           for st in plan.stages
       ) + "\n\n---\n\n## Definition of Done\n\nThis plan is complete when:\n- [ ] All stages are marked done\n"

       # Context file
       files[f"{root}/{n}-context/{n}-context.md"] = f"""# Plan {n} — Context Briefing

   ## Current State

   This plan has been generated by the Plan Builder UI and is ready to execute.

   ## Key Decisions Already Made

   - Plan type: {plan.type} ({{"d": "Development", "p": "Polishing", "r": "Refactoring"}}.get(plan.type, plan.type))
   - Total stages: {len(plan.stages)}
   - Total phases: {sum(len(st.phases) for st in plan.stages)}

   ## Constraints & Rules

   - Follow the stage/phase/part order exactly
   - Update the changelog after each completed part
   """

       # Changelog
       files[f"{root}/{n}-context/{n}-changelog.md"] = f"""# Plan {n} — Changelog

   ## Log

   _(No entries yet.)_
   """

       # Stage files
       for st in plan.stages:
           st_path = f"{root}/{n}-workflow/{n}-{st.id}"
           files[f"{st_path}/{n}-{st.id}.md"] = f"""# Stage {st.id.upper()} — {st.name}
   **Plan:** {n}

   ## Stage Goal

   {st.goal}

   ## Phases in This Stage

   | Phase | Name | Purpose | Status |
   |-------|------|---------|--------|
   """ + "\n".join(
               f"| {ph.id.upper()} | {ph.name} | {ph.goal} | Not started |"
               for ph in st.phases
           ) + "\n"

           # Phase files
           for ph in st.phases:
               ph_path = f"{st_path}/{ph.id}"
               files[f"{ph_path}/{ph.id}.md"] = f"""# {st.id.upper()} / {ph.id.upper()} — {ph.name}
   **Plan:** {n}

   ## Phase Goal

   {ph.goal}

   ## Parts in This Phase

   | Part | Name | Task | Status |
   |------|------|------|--------|
   """ + "\n".join(
                   f"| {pt.id.upper()} | {pt.name} | {pt.description} | Not started |"
                   for pt in ph.parts
               ) + "\n"

               # Part files
               for pt in ph.parts:
                   pt_path = f"{ph_path}/{pt.id}"
                   files[f"{pt_path}/{pt.id}.md"] = f"""# {st.id.upper()} / {ph.id.upper()} / {pt.id.upper()} — {pt.name}
   **Plan:** {n}

   ## Task

   {pt.description}

   ## Done When

   - [ ] Task is complete
   - [ ] No errors introduced
   - [ ] Changelog updated
   """

       return files
   ```

2. Create `src-python/routers/export_plan.py`:

   ```python
   from fastapi import APIRouter, HTTPException
   from pydantic import BaseModel
   from pathlib import Path
   from services.plan_builder import Plan, generate_plan_files

   router = APIRouter()

   class ExportRequest(BaseModel):
       plan: Plan
       output_dir: str  # absolute path to the plans/ folder

   class ExportResponse(BaseModel):
       files_written: int
       root_path: str

   @router.post("/export", response_model=ExportResponse)
   def export_plan(req: ExportRequest) -> ExportResponse:
       output_dir = Path(req.output_dir)
       if not output_dir.exists():
           raise HTTPException(status_code=400, detail=f"Output directory does not exist: {req.output_dir}")

       plan = req.plan
       type_folder = {"d": "d-plns", "p": "p-plns", "r": "r-plns"}.get(plan.type, "d-plns")
       target_dir = output_dir / type_folder

       files = generate_plan_files(plan)
       count = 0
       root_path = ""

       for rel_path, content in files.items():
           full_path = target_dir / rel_path
           full_path.parent.mkdir(parents=True, exist_ok=True)
           full_path.write_text(content, encoding="utf-8")
           count += 1
           if not root_path:
               root_path = str(full_path.parent.parent)

       return ExportResponse(files_written=count, root_path=root_path)
   ```
